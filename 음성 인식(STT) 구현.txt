Future<void> startListening() async {
  if (!_isListening) {
    _recognizedText = "";
    _isListening = true;
    notifyListeners();

    try {
      await speech.listen(
        onResult: (result) {
          _recognizedText = result.recognizedWords;
          notifyListeners();
        },
        listenFor: Duration(seconds: 30),
        pauseFor: Duration(seconds: 3),
        localeId: "ko_KR",
      );
    } catch (e) {
      // 오류 처리
    }
  }
}
